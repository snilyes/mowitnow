33              + source + "']");
34      List<String> champs = Splitter.on(SEPARATOR).splitToList(source);
35      int x = Integer.valueOf(champs.get(0)) - 1;
36      int y = Integer.valueOf(champs.get(1)) - 1;
37      Orientation orientation = Orientation.byCode(champs.get(2));
38      return new Mower(counter.getAndIncrement(), lawn.cellAt(x, y), orientation);
39    }
